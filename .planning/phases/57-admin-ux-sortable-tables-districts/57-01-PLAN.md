   ```typescript
   const searchParams = useSearchParams();
   const router = useRouter();
   const sortKey = searchParams.get("sort") ?? defaultSort ?? columns[0]?.key ?? "";
   const sortDir = (searchParams.get("dir") === "asc" ? "asc" : searchParams.get("dir") === "desc" ? "desc" : null) ?? defaultDir;
   const page = Math.max(0, Number(searchParams.get("page") ?? "0"));
   ```
3. Update `handleSort` to push URL changes:
   ```typescript
   function handleSort(key: string) {
     const params = new URLSearchParams(searchParams.toString());
     if (key === sortKey) {
       params.set("dir", sortDir === "asc" ? "desc" : "asc");
     } else {
       params.set("sort", key);
       params.set("dir", "desc");
     }
     params.delete("page"); // Reset pagination on sort change per Pitfall 1
     router.push(`?${params.toString()}`);
   }
   ```
4. Update pagination prev/next buttons to also use `router.push` with URL params:
   ```typescript
   function goToPage(newPage: number) {
     const params = new URLSearchParams(searchParams.toString());

       params.set("page", String(newPage));
     } else {
       params.delete("page");
     }
     router.push(`?${params.toString()}`);
   }
   ```
5. Remove the three useState lines (`sortKey`, `sortDir`, `page`).
6. Per D-02: Change `defaultDir` prop default from `"desc"` to `"asc"`. This makes every table default to ascending alphabetical sort when no URL params are present.
