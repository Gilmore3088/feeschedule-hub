       ```python
       def sweep_canonical_outliers(conn, fix=False):
           """Flag fees whose amount deviates 3+ stddev from canonical category median.
           
           Groups by canonical_fee_key (not fee_category) to catch outliers
           that are technically categorized correctly but have suspicious amounts.
           Skips categories with < 5 observations (insufficient statistical basis).
           Skips $0 amounts (legitimate "no charge" entries).
           """
           cur = conn.cursor()
           
           # Get per-canonical-key statistics
           cur.execute("""
               SELECT canonical_fee_key,
                      AVG(amount) as mean_amount,
                      STDDEV(amount) as stddev_amount,
                      PERCENTILE_CONT(0.5) WITHIN GROUP (ORDER BY amount) as median_amount,
                      COUNT(*) as obs_count
               FROM extracted_fees
               WHERE canonical_fee_key IS NOT NULL
                 AND amount IS NOT NULL
                 AND amount > 0
                 AND review_status != 'rejected'
               GROUP BY canonical_fee_key
               HAVING COUNT(*) >= 5
           """)
           stats = {row["canonical_fee_key"]: row for row in cur.fetchall()}
           
           # Find outliers: amount > 3 stddev from median
           flagged = []
           for key, s in stats.items():
               if s["stddev_amount"] is None or s["stddev_amount"] == 0:
                   continue
               threshold_low = s["median_amount"] - 3 * s["stddev_amount"]
               threshold_high = s["median_amount"] + 3 * s["stddev_amount"]
               
               cur.execute("""
                   SELECT id, fee_name, amount, review_status
                   FROM extracted_fees
                   WHERE canonical_fee_key = %s
                     AND amount IS NOT NULL
                     AND amount > 0
                     AND review_status NOT IN ('rejected', 'flagged')
                     AND (amount < %s OR amount > %s)
               """, (key, max(threshold_low, 0.01), threshold_high))
               
               for row in cur.fetchall():
                   flagged.append({
                       "fee_id": row["id"],
                       "fee_name": row["fee_name"],
                       "amount": row["amount"],
                       "canonical_key": key,
                       "median": s["median_amount"],
                       "stddev": s["stddev_amount"],
                   })
           
           # Report
           log.info(f"Canonical outlier sweep: {len(flagged)} outliers found across {len(stats)} categories")
           for f in flagged:
               log.info(f"  [{f['canonical_key']}] ${f['amount']:.2f} -- median ${f['median']:.2f}, stddev ${f['stddev']:.2f} -- {f['fee_name']}")
           
           # Fix mode: flag for review
           if fix and flagged:
               ensure_roomba_log(conn)
               for f in flagged:
                   cur.execute(
                       "UPDATE extracted_fees SET review_status = 'flagged' WHERE id = %s",
                       (f["fee_id"],)
                   )
                   log_change(
                       conn, f["fee_id"],
                       "review_status",
                       "pending",  # or whatever current status was
                       "flagged",
                       f"canonical_outlier: ${f['amount']:.2f} is 3+ stddev from {f['canonical_key']} median ${f['median']:.2f}"
                   )
               conn.commit()
               log.info(f"Flagged {len(flagged)} outlier fees for review")
           
           return flagged
       ```
