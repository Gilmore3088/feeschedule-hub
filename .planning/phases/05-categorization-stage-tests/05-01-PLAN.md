def run(
    db: Database,
    *,
    dry_run: bool = False,
    force: bool = False,
    limit: int | None = None,
) -> None:
    """Batch-categorize extracted fees using the fee name alias table.

    Reads extracted_fees WHERE fee_category IS NULL (unless force=True).
    Writes fee_category + fee_family for rows where normalize_fee_name()
    returns a canonical key in the _all_canonical() set.
    Rows whose name does not match any alias are left with fee_category=NULL.
    Calls emit_result() at the end — no return value.
    """
