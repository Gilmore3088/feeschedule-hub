import { HAMILTON_FORBIDDEN_PATTERNS, HAMILTON_FORBIDDEN_WORDS } from "./voice";

export interface ValidationViolation {
  number: number;
  context: string;          // ±30 chars around the violation in prose
}

export interface ValidationResult {
  valid: boolean;
  violations: ValidationViolation[];
  voiceViolations: string[];
  allowed_count: number;    // Total numbers in allowedSet
  found_count: number;      // Total numbers found in prose
}

const YEAR_RANGE = /^(19|20)\d{2}$/;

// Extract all numeric values from a string, handling comma separators and $ prefixes
export function extractNumbers(text: string): number[] {
  // Match: optional $, digits with optional commas, optional decimal
  const matches = text.matchAll(/\$?([\d,]+(?:\.\d+)?)\s*%?/g);
  const results: number[] = [];
  for (const m of matches) {
    const n = parseFloat(m[1].replace(/,/g, ""));
    if (!isNaN(n)) results.push(n);
  }
  return results;
}

// Extract all numbers from a deeply-nested data object (source of truth)
export function extractNumbersFromData(data: unknown): number[] {
  const numbers: number[] = [];
  function walk(val: unknown): void {
    if (typeof val === "number" && !isNaN(val)) {
      numbers.push(val);
    } else if (typeof val === "string") {
      extractNumbers(val).forEach(n => numbers.push(n));
    } else if (Array.isArray(val)) {
      val.forEach(walk);
    } else if (val !== null && typeof val === "object") {
      Object.values(val as Record<string, unknown>).forEach(walk);
    }
  }
  walk(data);
  return numbers;
}

function isYear(n: number): boolean {
  return Number.isInteger(n) && n >= 2000 && n <= 2030;
}

function numbersMatch(a: number, b: number): boolean {
  return Math.abs(a - b) <= 0.05;
}

export function validateNumbers(
  prose: string,
  allowedNumbers: number[]
): ValidationResult {
  const allowedSet = allowedNumbers;
  const proseNumbers = extractNumbers(prose);
  const violations: ValidationViolation[] = [];
  const voiceViolations: string[] = [];

  for (const n of proseNumbers) {
    if (isYear(n)) continue;                 // Allowlist years
    const allowed = allowedSet.some(a => numbersMatch(a, n));
    if (!allowed) {
      // Find context around this number
      const idx = prose.search(new RegExp(n.toString().replace(".", "\\.")));
      const start = Math.max(0, idx - 30);
      const end = Math.min(prose.length, idx + 30);
      violations.push({ number: n, context: prose.slice(start, end) });
    }
  }

  // Check forbidden words (case-insensitive, whole phrase)
  for (const word of HAMILTON_FORBIDDEN_WORDS) {
    if (prose.toLowerCase().includes(word.toLowerCase())) {
      voiceViolations.push(word);
    }
  }

  // Check forbidden patterns
  for (const pattern of HAMILTON_FORBIDDEN_PATTERNS) {
    if (pattern.test(prose)) {
      voiceViolations.push(pattern.source);
    }
  }

  return {
    valid: violations.length === 0 && voiceViolations.length === 0,
    violations,
    voiceViolations,
    allowed_count: allowedSet.length,
    found_count: proseNumbers.length,
  };
}
```

```typescript
import Anthropic from "@anthropic-ai/sdk";
import { HAMILTON_SYSTEM_PROMPT } from "./voice";
import { validateNumbers, extractNumbers, extractNumbersFromData } from "./validate-numbers";
import type { GenerateSectionInput, GenerateSectionOutput } from "./types";

const DEFAULT_MAX_WORDS: Record<string, number> = {
  monthly_pulse: 200,
};
const FALLBACK_MAX_WORDS = 400;

export class HamiltonValidationError extends Error {
  constructor(
    public readonly violations: Array<{ number: number; context: string }>,
    public readonly voiceViolations: string[]
  ) {
    const msg = violations.length > 0
      ? `Hamilton invented ${violations.length} number(s) not in source data: ${violations.map(v => v.number).join(", ")}`
      : `Hamilton voice violation: ${voiceViolations.join(", ")}`;
    super(msg);
    this.name = "HamiltonValidationError";
  }
}

export async function generateSection(
  input: GenerateSectionInput,
  // Injectable for testing — defaults to env-configured client
  client: Pick<Anthropic, "messages"> = new Anthropic()
): Promise<GenerateSectionOutput> {
  const maxWords = input.max_words
    ?? DEFAULT_MAX_WORDS[input.type]
    ?? FALLBACK_MAX_WORDS;

  const userPrompt = [
    `Write a ${maxWords}-word maximum ${input.type.replace(/_/g, " ")} section.`,
    input.tone_override ? `Tone guidance: ${input.tone_override}` : null,
    "",
    "Use only the figures provided below. Do not invent or estimate any number not present in this data.",
    "",
    "Source data:",
    "```json",
    JSON.stringify(input.data, null, 2),
    "```",
    "",
    "Return only the narrative prose. No section headers, no markdown, no lists — flowing paragraphs only.",
  ].filter(Boolean).join("\n");

  const response = await client.messages.create({
    model: "claude-sonnet-4-5",
    max_tokens: Math.ceil(maxWords * 1.5),   // tokens > words; leave headroom
    system: HAMILTON_SYSTEM_PROMPT,
    messages: [{ role: "user", content: userPrompt }],
  });

  const prose = response.content
    .filter(b => b.type === "text")
    .map(b => (b as { type: "text"; text: string }).text)
    .join("")
    .trim();

  // Numeric validation — zero tolerance for invented statistics (per D-17, D-18)
  const allowedNumbers = extractNumbersFromData(input.data);
  const validation = validateNumbers(prose, allowedNumbers);
  if (!validation.valid) {
    throw new HamiltonValidationError(validation.violations, validation.voiceViolations);
  }

  const numbersUsed = extractNumbers(prose);
  const words = prose.split(/\s+/).filter(Boolean);

  return {
    prose,
    word_count: words.length,
    numbers_used: numbersUsed,
    section_type: input.type,
    generated_at: new Date().toISOString(),
  };
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Source data JSON → Claude prompt | Data serialized with JSON.stringify — no injection vectors since data comes from internal DB queries |
| Claude response prose → validator | Claude output is untrusted until validateNumbers confirms all numbers exist in source data |
| HamiltonValidationError → caller | Caller (report template) must handle this error — uncaught → report job fails (correct behavior) |

| T-12-05 | Information Disclosure | source data in user prompt | accept | Data comes from internal DB; prompt is server-side only; never logged to client |
