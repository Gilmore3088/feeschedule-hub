```sql
-- hamilton_signals: immutable signal records
CREATE TABLE IF NOT EXISTS hamilton_signals (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  institution_id TEXT NOT NULL,
  signal_type TEXT NOT NULL,
  severity TEXT NOT NULL,
  title TEXT NOT NULL,
  body TEXT NOT NULL,
  source_json JSONB,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- hamilton_priority_alerts: user-scoped alert instances
CREATE TABLE IF NOT EXISTS hamilton_priority_alerts (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id INTEGER NOT NULL,
  signal_id UUID NOT NULL REFERENCES hamilton_signals(id),
  status TEXT NOT NULL DEFAULT 'active' CHECK (status IN ('active', 'acknowledged', 'dismissed')),
  acknowledged_at TIMESTAMPTZ,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);
```
