    After the existing `listSavedAnalyses` export, add:
    ```typescript
    /**
     * Load a single saved analysis by ID for the current user.
     * Returns the stored AnalyzeResponse or null if not found/unauthorized.
     */
    export async function loadAnalysis(id: string): Promise<AnalyzeResponse | null> {
      const user = await getCurrentUser();
      if (!user) return null;

      try {
        const rows = await sql<Array<{ response_json: string }>>`
          SELECT response_json::text
          FROM hamilton_saved_analyses
          WHERE id = ${id}::uuid
            AND user_id = ${user.id}
            AND status = 'active'
          LIMIT 1
        `;
        if (!rows[0]) return null;
        return JSON.parse(rows[0].response_json) as AnalyzeResponse;
      } catch {
        return null;
      }
    }
    ```

    Note: Use `response_json::text` in SELECT and `JSON.parse` because the postgres driver may return the JSONB column as a string. The `::uuid` cast on `id` is required because the column is UUID type (confirmed by existing `RETURNING id::text` in saveAnalysis). Auth is scoped by `user_id = ${user.id}` — no additional premium check needed since the analysis was already saved by a premium user.
