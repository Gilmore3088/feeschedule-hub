def determine_review_status(
    flags: list[ValidationFlag],
    confidence: float,
    config: Config,
    fee_category: str | None = None,
    amount: float | None = None,

    """
    Returns 'flagged' if any error or warning flags exist (has_blocking=True).

         ```python
         def flag_amount_outliers(db: Database, config: Config) -> int:
             """Flag extracted_fees rows with amounts 3+ std dev from their category median.

             Operates on fees with review_status != 'rejected' and fee_category IS NOT NULL
             and amount IS NOT NULL.

             Returns count of rows flagged.
             """
             import json, statistics
             threshold = config.extraction.outlier_std_dev_threshold  # 3.0
             rows = db.fetchall(
                 """SELECT id, fee_category, amount, validation_flags
                    FROM extracted_fees
                    WHERE fee_category IS NOT NULL AND amount IS NOT NULL
                    AND review_status != 'rejected'"""
             )
             # Group amounts by category
             by_cat: dict[str, list[tuple[int, float, str | None]]] = {}
             for r in rows:
                 by_cat.setdefault(r["fee_category"], []).append(
                     (r["id"], r["amount"], r["validation_flags"])
                 )

             flagged = 0
             for cat, items in by_cat.items():
                 if len(items) < 3:
                     continue
                 amounts = [amt for _, amt, _ in items]
                 mean = statistics.mean(amounts)
                 stdev = statistics.stdev(amounts)
                 if stdev == 0:
                     continue
                 for fee_id, amt, existing_flags in items:
                     z = abs(amt - mean) / stdev
                     if z >= threshold:
                         flag = {"rule": "statistical_outlier", "severity": "warning",
                                 "message": f"Amount ${amt:.2f} is {z:.1f} std dev from category mean ${mean:.2f}"}
                         existing = json.loads(existing_flags) if existing_flags else []
                         existing.append(flag)
                         db.execute(
                             "UPDATE extracted_fees SET validation_flags = ? WHERE id = ?",
                             (json.dumps(existing), fee_id),
                         )
                         flagged += 1
             db.commit()
             return flagged
         ```
