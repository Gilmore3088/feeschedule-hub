```typescript
  trend: 'rising' | 'falling' | 'stable';
  asOf: string;
}

// Currently NOT exported -- Task 1 must add `export` keyword:
function deriveTrend(
  current: number,
  history: { value: number }[]
): 'rising' | 'falling' | 'stable' {
  if (history.length < 2) return 'stable';
  const oldest = history[0].value;
  if (oldest === 0) return 'stable';
  const diffPct = ((current - oldest) / Math.abs(oldest)) * 100;
  if (diffPct > 0.5) return 'rising';
  if (diffPct < -0.5) return 'falling';
  return 'stable';
}
```

```typescript
function priorYearQuarter(quarter: string): string {
  const match = /^(\d{4})-Q(\d)$/.exec(quarter);
  if (!match) return "";
  const year = parseInt(match[1], 10) - 1;
  return `${year}-Q${match[2]}`;
}
```
