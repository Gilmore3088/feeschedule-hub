```tsx
async function getRelatedReports(inst: { charter_type: string; asset_size_tier: string | null; fed_district: number | null }): Promise<{ slug: string; title: string }[]> {
  try {
    const { sql: getSql } = await import("@/lib/crawler-db/connection");
    const sqlClient = getSql();
    const rows = await sqlClient<{ slug: string; title: string }[]>`
      SELECT slug, title
      FROM published_reports
      WHERE is_public = true
      ORDER BY published_at DESC
      LIMIT 20
    `;
    // Filter client-side by matching keywords in title (charter, district, tier)
    const keywords: string[] = [];
    if (inst.charter_type) keywords.push(inst.charter_type === "bank" ? "bank" : "credit union");
    if (inst.fed_district) keywords.push(`district ${inst.fed_district}`);
    if (inst.asset_size_tier) keywords.push(inst.asset_size_tier);

    const matched = rows.filter(r => {
      const titleLower = r.title.toLowerCase();
      return keywords.some(kw => titleLower.includes(kw.toLowerCase()));
    });
    return matched.slice(0, 5);
  } catch {
    return [];
  }
}
```
