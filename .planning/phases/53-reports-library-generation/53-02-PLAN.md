```typescript
useEffect(() => {
  if (!initialScenarioId) return;
  let cancelled = false;
  loadScenarioById(initialScenarioId).then((scenario) => {
    if (cancelled || !scenario) return;
    // Auto-select Category Deep Dive template (natural fit for single-category scenario)
    setSelectedTemplate("category_deep_dive");
    // Set focus area to the scenario's fee category (display-friendly)
    setFocusArea(scenario.fee_category.replace(/_/g, " "));
  });
  return () => { cancelled = true; };
}, [initialScenarioId]);
```
