    ```typescript
    export async function generatePresignedUrl(key: string, ttlSeconds: number): Promise<string> {
      const client = new S3Client({
        endpoint: process.env.R2_ENDPOINT!,
        region: 'auto',
        credentials: {
          accessKeyId: process.env.R2_ACCESS_KEY_ID!,
          secretAccessKey: process.env.R2_SECRET_ACCESS_KEY!,
        },
      })
      const cmd = new GetObjectCommand({ Bucket: process.env.R2_BUCKET ?? 'bfi-reports', Key: key })
      return getSignedUrl(client, cmd, { expiresIn: ttlSeconds })
    }
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| HTTP caller → generate route | report_type and params are untrusted; validated before DB insert |
| Status/download routes → report_jobs | Row-level ownership enforced in application layer |
| Presigned URL → R2 | Short TTL (1h) limits sharing window; URL is a bearer token |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-13-10 | Spoofing | POST /api/reports/generate | mitigate | getCurrentUser() verifies session before any processing; 401 on null user |
| T-13-11 | Information Disclosure | GET /api/reports/[id]/status | mitigate | Ownership check enforced: users can only view their own jobs; admin can view cron jobs (user_id IS NULL) |
| T-13-12 | Elevation of Privilege | GET /api/reports/[id]/download | mitigate | Same ownership guard as status route; presigned URL generated only after auth passes |
| T-13-13 | Tampering | report_type in POST body | mitigate | Explicit allowlist validation against ReportType union before INSERT; SQL uses parameterized values |
| T-13-14 | Denial of Service | Rapid polling of status route | accept | No rate limiting at this layer — Modal function is idempotent; job table has indexed id column; acceptable for admin-only v2.0 scope |
</threat_model>
